\macro zapfd

\newdimen\zdboxwidth
\zdboxwidth=\hsize
\divide\zdboxwidth by 8
\advance\zdboxwidth-\framethicknessright
\advance\zdboxwidth-\framethicknessleft
\advance\zdboxwidth-\framemarginleft
\advance\zdboxwidth-\framemarginright


\def\makezdchar#1{%
  \ftbox \the\zdboxwidth {%
    \parindent 0pt
    \toksloadoctal#1\to\rtA
    {\setfontsize{5}\it'\the\rtA}%
    \hfil
    {\setfontsize{32}\zdchar#1%
    %
    % Un `strut' ad-hoc
    %
    \vrule height 28pt depth 5pt width 0em%
    }%
    \hfil
  }%
}

\offinterlineskip
\noindent
\rcA=0
\loop
  \makezdchar\rcA
  \penalty0 %
\ifnum\rcA<255\relax
  \advance\rcA by 1\relax
\repeat

\bye
